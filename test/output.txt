import uuid

def remove_dashes_and_generate_uid(input_string):
    # Remove dashes from the input string
    cleaned_string = input_string.replace('-', '')

    # Generate a UID using the cleaned string
    generated_uid = uuid.uuid5(uuid.NAMESPACE_DNS, cleaned_string)

    return str(generated_uid)

# Example usage:
input_string = "a-b-c-d-1234"
result_uid = remove_dashes_and_generate_uid(input_string)
print(result_uid)
